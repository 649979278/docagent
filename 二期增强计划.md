WorkAgent 二期详细落地方案

 Context

 WorkAgent 当前有"本地 Ollama 模型 + Electron 壳 + 自研 AgentRuntime + 文档导入/RAG"的雏形，但运行时行为与 Claude Code 差距显著：turn state 初始化时序错误导致路由不可靠、RAG 注入在 budget
 之外 splice、工具全量串行无并发编排、Compact 没有 boundary 和恢复机制、Router 基于关键词而非状态机。这些不是模型问题，是架构行为问题——运行时没有把模型"约束在正确轨道上"。

 本方案以运行时行为对齐 Claude Code为最高优先级，按"先修核心 Bug → 重构运行时 → 升级外围"的顺序，7 个迭代周期推进。增量拆分，不推翻现有逻辑。诊断数据写入 DB 并在右侧 UI 抽屉展示。

 1.1 修复 turn state 初始化时序

 问题：loopState.currentTurn 在 routeAfterResponse() 之后才赋值（runtime.ts:358-361），导致 router 检查 state.currentTurn?.hasError 时永远是 null。

 修改文件：packages/agent-core/src/runtime.ts

 改动要点：
 1. 在 while 循环顶部立即创建 TurnState，赋值给 loopState.currentTurn
 2. 模型调用后立即更新 assistantContent/toolCalls/tokensUsed
 3. 工具执行后立即设置 hasError（当前缺失）
 4. routeAfterResponse() 调用在 turn state 完整后执行

 router.ts 重构核心：
 export function routeAfterResponse(
   state: LoopState,
   hasToolCalls: boolean,
   activePlan: Plan | null,
 ): RouteResult {
   // 1. 错误优先
   if (state.currentTurn?.hasError && !state.currentTurn.retried) {
     return { decision: 'Continue', transition: 'error_retry', reason: '工具失败，重试一次' };
   }

   // 2. 有工具调用 → 继续执行
   if (hasToolCalls) {
     // chat mode 下不自动切 plan，只发建议事件
     return { decision: 'Continue', transition: 'next_turn', reason: '有工具调用待执行' };
   }

   // 3. 按 mode 路由（不再用关键词判断）
   switch (state.mode) {
     case 'plan':
       return routePlanMode(state, activePlan);
     case 'execute':
       return routeExecuteMode(state, activePlan);
     default:
       return { decision: 'Done', transition: 'completed', reason: '模型已生成最终回复' };
   }
 }

 // routePlanMode 基于状态而非关键词
 function routePlanMode(state: LoopState, activePlan: Plan | null): RouteResult {
   if (activePlan?.status === 'draft') {
     return { decision: 'WaitApproval', transition: 'wait_approval', reason: '计划已生成', phaseSwitch: 'PLAN_REVIEW' };
   }
   if (activePlan?.status === 'approved') {
     return { decision: 'EnterPlan', transition: 'execute_plan', reason: '计划已批准', modeSwitch: 'execute', phaseSwitch: 'EXECUTE_DRAFT' };
   }
   if (activePlan?.status === 'executing') {
     return { decision: 'Continue', transition: 'next_turn', reason: '计划执行中' };
   }
   return { decision: 'Continue', transition: 'next_turn', reason: '继续收集信息', phaseSwitch: 'PLAN_COLLECT' };
 }

 async *runTurn(sessionId: string, input: string, mode: AgentMode = 'chat'): AsyncGenerator<AgentEventEnvelope> {
   // 初始化
   let state = createQueryLoopState(sessionId, mode, budget, memories);
   state.messages = this.buildMessages(recentMessages, input, mode, memories, budget);

   // RAG 增强移入 pipeline（见迭代3详细实现）
   state = await this.enrichWithContext(state, input);

   let loopCount = 0;
   while (state.transition !== 'completed' &&
          state.transition !== 'aborted' &&
          state.transition !== 'max_turns' &&
          state.transition !== 'prompt_too_long' &&
          loopCount < this.config.maxTurns) {
     loopCount++;

     // 1. compact boundary 切片
     const messagesAfterBoundary = getMessagesAfterCompactBoundary(state);

     // 2. 工具结果预算
     applyToolResultBudget(messagesAfterBoundary, state.budget);

     // 3. 微压缩
     const microResult = microCompact(messagesAfterBoundary);

     // 4. 上下文折叠（合并连续 assistant）
     const collapsedMessages = contextCollapse(microResult);

     // 5. 自动压缩评估
     state = await this.maybeAutoCompact(state, collapsedMessages);

     // 6. token 阻塞限制
     const estimatedTokens = estimateMessagesTokens(state.messages);
     if (estimatedTokens > state.budget.total * 0.95) {
       state = { ...state, transition: 'prompt_too_long' };
       break;
     }

     // 7. 调用模型
     const modelResult = await this.callModel(state);

     // 8-9. 工具执行
     state = await this.executeToolsAndUpdate(state, modelResult);

     // 10. 持久化
     this.persistMessages(sessionId, turnId, state.messages, existingMessageIds);

     // 11. token 预算检查（收益递减检测）
     state = this.checkTokenBudget(state, modelResult.usage);

     // 12. 不可变更新
     state = { ...state, turnCount: state.turnCount + 1 };
   }

   // 产出完成事件
   yield this.createEvent(sessionId, turnId, sequence++, 'done', null);
 }

 /** 获取 compact boundary 之后的消息 */
 export function getMessagesAfterCompactBoundary(state: QueryLoopState): Message[] {
   if (!state.lastCompactBoundaryId) return state.messages;
   const idx = state.messages.findIndex(m => m.compactBoundaryId === state.lastCompactBoundaryId);
   return idx >= 0 ? state.messages.slice(idx + 1) : state.messages;
 }

 /** 应用工具结果预算 - 截断超出预算的工具输出 */
 export function applyToolResultBudget(messages: Message[], budget: ContextBudget): Message[] {
   let toolTokensUsed = 0;
   return messages.map(msg => {
     if (msg.role === 'tool') {
       if (toolTokensUsed + msg.tokenCount > budget.toolResults) {
         return { ...msg, content: `[结果已压缩: ${msg.toolCallId}]`, tokenCount: 5 };
       }
       toolTokensUsed += msg.tokenCount;
     }
     return msg;
   });
 }

 /** 上下文折叠 - 合并连续 assistant 消息（无工具调用的） */
 export function contextCollapse(messages: Message[]): Message[] {
   const result: Message[] = [];
   for (const msg of messages) {
     const last = result[result.length - 1];
     if (last && last.role === 'assistant' && msg.role === 'assistant' && !msg.toolCalls?.length) {
       result[result.length - 1] = { ...last, content: last.content + '\n' + msg.content, tokenCount: last.tokenCount + msg.tokenCount };
     } else {
       result.push({ ...msg });
     }
   }
   return result;
 }

 /** Token 预算检查 - 收益递减检测 */
 export function checkTokenBudget(
   state: QueryLoopState,
   usage: { promptTokens: number; completionTokens: number; totalTokens: number },
 ): QueryLoopState {
   // 简化实现：如果 completionTokens < 500 且 turnCount > 3 → 停止
   if (state.turnCount > 3 && usage.completionTokens < 500) {
     return { ...state, transition: 'completed' };
   }
   return state;
 }

 export interface PartitionedCalls {
   readOnlyCalls: ToolCall[];
   writeCalls: ToolCall[];

 /** 将工具调用分为只读批和写入批 - 参考 Claude Code partitionToolCalls */
 export function partitionToolCalls(calls: ToolCall[], registry: ToolRegistry): PartitionedCalls {
   const readOnlyCalls: ToolCall[] = [];
   const writeCalls: ToolCall[] = [];
   for (const call of calls) {
     const tool = registry.getTool(call.name);
     if (tool && tool.isReadOnly()) {
       readOnlyCalls.push(call);
     } else {
       writeCalls.push(call);
     }
   }
   return { readOnlyCalls, writeCalls };
 }

 async executeAll(calls: ToolCall[], context: ToolContext): Promise<ToolExecutionResult[]> {
   const { readOnlyCalls, writeCalls } = partitionToolCalls(calls, this.registry);

   // 1. 只读工具并发执行（max concurrency 10）
   const readOnlyResults = await this.executeConcurrent(readOnlyCalls, context, 10);

   let index = 0;

   const runNext = async (): Promise<void> => {
     while (index < calls.length) {
       const i = index++;
       const result = await this.executeOne(calls[i], context);
       results.set(calls[i].id, result);
     }
   };

   const workers = Math.min(maxConcurrency, calls.length);
   await Promise.all(Array.from({ length: workers }, () => runNext()));
   return calls.map(c => results.get(c.id)!);
 }

 保留原有串行路径：通过 config.enableConcurrent = false 可回退到全量串行。

 // docx.ts 改造骨架
 import { unzipSync } from 'fflate';

 /** DOCX 文档解析器 - 使用 fflate 替代手写 ZIP 解析 */
 export class DocxExtractor implements DocumentExtractor {
   async extract(filePath: string): Promise<ExtractedDocument> {
     const buffer = await fs.readFile(filePath);
     const entries = unzipSync(new Uint8Array(buffer));

     // 解析 word/document.xml
     const documentXml = entries['word/document.xml'];
     if (!documentXml) throw new Error('无效的DOCX文件');

     // 使用 fast-xml-parser 处理 XML（替代正则）
     const { sections, outline } = this.parseDocumentXml(documentXml);

     return {
       filePath,
       fileName: path.basename(filePath),
       fileType: 'docx',
       content: sections.map(s => s.content).join('\n'),
       sections,
       metadata: {
         contentHash: this.computeHash(buffer),
         outline,
         sectionCount: sections.length,
         paragraphCount: sections.reduce((sum, s) => sum + s.paragraphCount, 0),
       },
     };
   }

   /** 使用 XML 解析器处理 word/document.xml */
   private parseDocumentXml(xmlBuffer: Uint8Array): { sections: DocumentSection[]; outline: string[] } {
     const parser = new XMLParser({ ignoreAttributes: false });
     const xml = parser.parse(xmlBuffer.toString('utf-8'));
     const body = xml['w:document']?.['w:body'];

     const sections: DocumentSection[] = [];
     const outline: string[] = [];

     for (const element of body?.['w:p'] ?? []) {
       // 解析段落样式确定标题层级
       const pStyle = element?.['w:pPr']?.['w:pStyle']?.['@_w:val'];
       const isHeading = pStyle?.startsWith('Heading');
       const level = isHeading ? parseInt(pStyle.replace('Heading', '')) : 0;

       // 提取文本
       const text = this.extractParagraphText(element);
       if (!text) continue;

       if (isHeading && level > 0) {
         outline.push(text);
         sections.push({ title: text, content: text, level, locator: `段落${sections.length + 1}` });
       } else if (sections.length > 0) {
         sections[sections.length - 1].content += '\n' + text;
       } else {
         sections.push({ title: '', content: text, level: 0, locator: `段落${sections.length + 1}` });
       }
     }

     return { sections, outline };
   }
 }

 6.4 RAG 检索升级：retrieval-pipeline

 新增文件：packages/rag/src/retrieval-pipeline.ts

 /** 检索管道 - 参考 Haystack 的可组合管道模式 */
 export class RetrievalPipeline {
   private index: KnowledgeIndex;
   private embedder: OllamaEmbedder;

   constructor(index: KnowledgeIndex, embedder: OllamaEmbedder) {
     this.index = index;
     this.embedder = embedder;
   }

   /** 执行检索流程：query normalize → retrieve → filter → pack citations */
   async retrieve(query: string, options: SearchOptions = {}): Promise<KnowledgeSearchResponse> {
     // 1. Query normalize（去除前后空格、统一全半角）
     const normalizedQuery = this.normalizeQuery(query);

     // 2. Embed query
     const queryVector = await this.embedder.embed(normalizedQuery);

     // 3. Vector search
     const rawResults = await this.index.search(queryVector, {
       topK: (options.topK ?? 5) * 2,  // 扩大检索范围，后续过滤
       metadataFilter: options.metadataFilter,
     });

     // 4. minScore 过滤
     const filtered = rawResults.filter(chunk => chunk.score >= (options.minScore ?? 0.3));

     // 5. budgetTokens 截断
     let budgeted = filtered;
     if (options.budgetTokens) {
       const result = truncateRagChunks(filtered, options.budgetTokens);
       budgeted = result.chunks;
     }

     // 6. Pack citations（带 [ref_N] 引用标签）
     const results = budgeted.slice(0, options.topK ?? 5);

     // 7. 记录检索结果到 agent_events（rag_enrich 事件）
     return {
       query: normalizedQuery,
       topK: options.topK ?? 5,
       results,
       error: results.length === 0 ? '未找到相关内容' : undefined,
     };
   }

   private normalizeQuery(query: string): string {
     return query.trim()
       .replace(/[！-～]/g, c => String.fromCharCode(c.charCodeAt(0) - 0xFEE0))  // 全角→半角
       .replace(/\s+/g, ' ');
   }
 }
