离线公文写作Agent — 一期最终落地方案

 Context

 离线公文写作桌面Agent。Windows用户安装后可本地对话、检索知识库、解析文件、Plan模式写作、输出.docx。本地Ollama部署模型（qwen3.5:9b + bge-m3），后续可接局域网满血模型。

 架构决策：Electron + TypeScript统一架构，参考Claude Code的Agent/Tool/Compact/Plan核心模式，参考AnythingLLM/Open WebUI/Jan的产品边界。

 4.1 AgentRuntime

 // packages/agent-core/src/runtime.ts

 /**
  * Agent运行时 - 参考Claude Code query.ts模式
  * 一期工具串行执行，每turn有turn_id，所有事件可持久化
  */
 export class AgentRuntime {
   async *runTurn(
     sessionId: string,
     input: UserInput,
     mode: AgentMode = 'chat',
   ): AsyncGenerator<AgentEventEnvelope, void, void> {
     const state = await this.loadState(sessionId);
     const turnId = generateId();
     let seq = 0;
     const emit = (type: AgentEventType, data: unknown) =>
       ({ type, data, sessionId, turnId, sequence: seq++, createdAt: Date.now() });

     while (true) {
       try {
         // 1. 上下文压缩
         const compactResult = await this.contextManager.compact(state);
         if (compactResult.freedTokens > 0) yield emit('compact', compactResult);

         // 2. 消息增强（意图检测+记忆注入）
         const enriched = await this.contextManager.enrich(state, input);

         // 3. 按mode+权限选择工具
         const tools = this.toolRegistry.getTools(state.context.mode, state.context.permissions);

         // 4. 预算分配
         const budget = this.contextManager.allocateBudget(state, tools);

         // 5. 调用Ollama流式
         const stream = this.modelProvider.chat({
           messages: enriched.messages,
           tools,
           model: this.getModelForMode(state.context.mode),
         });

         // 6. 处理流式响应
         const toolCalls: ToolCall[] = [];
         let textContent = '';
         for await (const event of stream) {
           if (event.type === 'token') {
             textContent += event.data;
             yield emit('token', event.data);
           } else if (event.type === 'tool_call') {
             toolCalls.push(event.toolCall);
           } else if (event.type === 'usage') {
             state.tokenCount = event.data.prompt_tokens; // ground truth
           }
         }

         // 7. 无工具调用→结束
         if (toolCalls.length === 0) {
           state.messages.push(createAssistantMessage(textContent));
           await this.persistTurn(state);
           yield emit('done', null);
           return;
         }

         // 8. 串行执行工具
         for (const toolCall of toolCalls) {
           const tool = this.toolRegistry.getTool(toolCall.name);
           yield emit('tool_start', { name: tool.name, input: toolCall.arguments });

           // 权限检查
           const permission = await this.permissionBroker.check(tool, toolCall.arguments, state);
           if (!permission.allowed) {
             yield emit('permission_denied', { toolName: tool.name, reason: permission.reason });
             continue;
           }

           // 执行（带超时）
           try {
             const result = await withTimeout(
               tool.call(toolCall.arguments, { state, modelProvider: this.modelProvider }),
               60_000,
             );
             const summary = tool.renderSummary(result);
             state.messages.push(createToolResultMessage(toolCall.id, result, summary));
             yield emit('tool_result', { name: tool.name, output: result });
           } catch (error) {
             // 工具失败：一次fallback解析，不无限重试
             const fallback = this.parseToolCallFallback(toolCall, error);
             if (fallback) {
               state.messages.push(createToolResultMessage(toolCall.id, fallback));
               yield emit('tool_result', { name: tool.name, output: fallback, isError: true });
             } else {
               yield emit('recoverable_error', { code: 'TOOL_FAILED', message: String(error) });
             }
           }
         }

         // 9. 持久化turn
         await this.persistTurn(state);
         // 继续下一轮
       } catch (error) {
         const recovery = this.handleError(error);
         if (recovery.action === 'retry') continue;
         yield emit('error', { code: recovery.code, message: recovery.userMessage });
         return;
       }
     }
   }
 }

 4.3 PlanModeController（驱动执行，不暴露plan_execute给模型）

 // packages/agent-core/src/plan-controller.ts

 /**
  * Plan模式状态机 - 公文写作领域化
  * 模型不能自行决定是否执行，必须用户批准后由Controller驱动
  */
 export type PlanPhase =
   | 'PLAN_COLLECT'   // 收集文种、对象、单位、材料、篇幅、风格
   | 'PLAN_RESEARCH'  // 只读检索和材料研读
   | 'PLAN_DRAFT'     // 生成计划和提纲
   | 'PLAN_REVIEW'    // 用户审查/编辑/批准
   | 'EXECUTE_DRAFT'  // 生成Markdown草稿
   | 'EXECUTE_EXPORT';// 用户确认后导出docx

 export class PlanModeController {
   private phase: PlanPhase = 'PLAN_COLLECT';

   /**
    * 根据当前phase决定可用工具和Agent行为
    */
   getToolsForPhase(registry: ToolRegistry): AgentTool[] {
     switch (this.phase) {
       case 'PLAN_COLLECT':
       case 'PLAN_RESEARCH':
         return registry.getTools('plan'); // 只读+plan工具
       case 'PLAN_DRAFT':
         return registry.getTools('plan');
       case 'PLAN_REVIEW':
         return []; // 等待用户操作
       case 'EXECUTE_DRAFT':
         return registry.getTools('execute'); // 允许write_output
       case 'EXECUTE_EXPORT':
         return [registry.getTool('doc_write'), registry.getTool('doc_overwrite')];
     }
   }

   /** 用户批准后进入执行阶段 */
   approvePlan(plan: Plan): void {
     if (this.phase !== 'PLAN_REVIEW') return;
     this.phase = 'EXECUTE_DRAFT';
     // 不由模型决定执行，由Controller驱动
   }

   /** 用户取消 → 回到CHAT */
   cancelPlan(): void {
     this.phase = 'PLAN_COLLECT'; // 重置
   }
 }

 4.4 上下文压缩（不做盲截断）

 // packages/agent-core/src/context/compact.ts

 /**
  * 上下文压缩 - 一期策略
  * 1. microCompact: 清理重复tool_result、缩短长工具输出、保留引用ID
  * 2. summaryCompact: 超过75%时对旧消息生成结构化摘要
  * 3. citationRehydrate: 生成正文前根据引用ID重新取回关键RAG片段
  * 4. reactiveCompact: Ollama报context超限时立即summaryCompact后重试
  *
  * 摘要必须保留：用户明确要求、文种格式约束、当前计划、已选材料来源、禁用表达、已生成草稿关键结论
  */
 export async function compactContext(state: AgentState): Promise<CompactResult> {
   // Level 1: microCompact - 每轮执行
   state.messages = microCompact(state.messages);

   const usage = estimateTokenUsage(state);
   const contextLength = await getContextLength(); // 从Ollama /api/show获取

   if (usage.total > contextLength * 0.75) {
     // Level 2: summaryCompact - 结构化摘要
     const summary = await summaryCompact(state.messages, {
       preserveTags: ['user_requirement', 'plan', 'citation', 'format_constraint', 'banned_expression'],
       keepRecentMessages: 6,
     });
     state.messages = [createSummaryMessage(summary), ...state.messages.slice(-6)];
     return { freedTokens: usage.total - estimateTokenUsage(state), level: 2 };
   }

   return { freedTokens: 0, level: 1 };
 }

 /** 引用重载 - 生成docx前根据计划中的引用ID重新取回完整RAG片段 */
 export async function citationRehydrate(
   plan: Plan,
   knowledgeIndex: KnowledgeIndex,
 ): Promise<Map<string, RetrievedChunk>> {
   const citationIds = plan.outline.citations ?? [];
   const chunks = new Map<string, RetrievedChunk>();
   for (const id of citationIds) {
     const chunk = await knowledgeIndex.getByChunkId(id);
     if (chunk) chunks.set(id, chunk);
   }
   return chunks;
 }

 // packages/agent-core/src/context/budget.ts

 /** 预算式上下文分配 - 防止某个区域膨胀饿死其他 */
 export function allocateBudget(contextLength: number, mode: AgentMode): ContextBudget {
   const systemPrompt = 500;
   const maxCompletion = 4096;
   const available = contextLength - systemPrompt - maxCompletion;

   if (mode === 'plan') {
     // Plan模式需要更多RAG空间
     return {
       systemPrompt,
       conversationHistory: Math.floor(available * 0.40),
       ragResults: Math.floor(available * 0.45),
       toolResults: Math.floor(available * 0.15),
       maxCompletionTokens: maxCompletion,
     };
   }
   return {
     systemPrompt,
     conversationHistory: Math.floor(available * 0.60),
     ragResults: Math.floor(available * 0.25),
     toolResults: Math.floor(available * 0.15),
     maxCompletionTokens: maxCompletion,
   };
 }

 /** 每类错误的恢复策略 */
 export function getRecoveryStrategy(error: AgentError): RecoveryAction {
   switch (error.code) {
     case 'OLLAMA_UNAVAILABLE': return { action: 'retry', interval: 5000, maxRetries: 12 };
     case 'CONTEXT_LENGTH_EXCEEDED': return { action: 'compact_and_retry', maxRetries: 3 };
     case 'MODEL_NOT_AVAILABLE': return { action: 'prompt_pull_model' };
     case 'TOOL_EXECUTION_FAILED': return { action: 'report_and_offer_retry' };
     case 'PERMISSION_DENIED': return { action: 'abort' };
     default: return { action: 'abort' };
   }
 }

 export class OllamaLifecycle {
   async initialize(): Promise<OllamaStatus> {
     // 1. 尝试连接已运行的Ollama
     if (await this.isOllamaRunning()) return { status: 'running' };

     // 2. 检查安装
     const ollamaPath = await this.findOllamaBinary();
     if (!ollamaPath) return { status: 'not_installed' }; // 显示引导，不静默安装

     // 3. 尝试启动ollama serve
     const process = spawn(ollamaPath, ['serve'], { detached: true, stdio: 'ignore' });
     process.unref();

     // 4. 等待就绪(最多30s)
     await this.waitForReady(30_000);
     this.startHealthMonitoring();
     return { status: 'started' };
   }

     setInterval(async () => {
       if (this.ollamaPid && !isProcessRunning(this.ollamaPid)) this.handleCrash();
       if (!(await this.isOllamaRunning())) this.handleCrash();
     }, 30_000);
   }

   /** 崩溃重启(指数退避，最多3次) */

 }
